#include <stdio.h>
int main() {
    double num1, num2;
    printf("Enter the first number: ");
    scanf("%lf", &num1);
    printf("Enter the second number: ");
    scanf("%lf", &num2);
    if (num2 != 0) {
        double quotient = num1 / num2;
        printf("Quotient: %.2lf\n", quotient);
    } else {
        printf("Error: Division by zero is not allowed.\n");
    }
return 0;
}
